 public class HomeController : Controller
    {
        AppleCompanyEntities db = new AppleCompanyEntities();

        // GET: Home
        public ActionResult Index(string search,string Cities)
        {
            if (Cities == "Karachi")
            {         
                var data = db.EmployeesTables.Where(model => model.City == "Karachi" && model.Name.StartsWith(search)).ToList();
                return View(data);
            }
            else if (Cities == "Lahore")
            {
                var data = db.EmployeesTables.Where(model => model.City == "Lahore" &&model.Name.StartsWith(search)).ToList();
                return View(data);
            }
            else if (Cities == null)
            {
                var data = db.EmployeesTables.ToList();
                return View(data);
            }
            else
            {
                var data = db.EmployeesTables.Where(model => model.Name.StartsWith(search)).ToList();           
                return View(data);
            }
        }
